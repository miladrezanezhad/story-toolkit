API Documentation

Complete reference for all classes, methods, and parameters — Story Development Toolkit v1.0.0

Python 3.8+ v1.0.0 MIT

Installation

bash 
git clone https://github.com/miladrezanezhad/story-toolkit.git
cd story-toolkit
pip install -r requirements.txt

StoryToolkit

Main facade class providing unified access to all toolkit features.

Constructor

python
from story_toolkit import StoryToolkit

toolkit = StoryToolkit()

Attributes

AttributeTypeDescription
engineStoryEngineCore story engine
character_genCharacterGeneratorCharacter generator
plot_genPlotGeneratorPlot generator
dialogue_genDialogueGeneratorDialogue generator
coherence_checkerCoherenceCheckerCoherence analyzer
text_analyzerTextAnalyzerText analysis tool

Methods

create_story(genre, theme, complexity=3) → dict

Creates a new story framework with all necessary components.

ParameterTypeDefaultDescription
genrestrRequiredfantasy, mystery, romance, adventure, sci_fi
themestrRequirede.g., courage, redemption, identity
complexityint3Complexity level 1–5

Returns: dict with metadata, outline, plot, characters, world, dialogue_scenes, coherence_report

python
story = toolkit.create_story(genre="fantasy", theme="courage", complexity=4)
print(story["metadata"]["genre"])  # "fantasy"
add_character_to_story(story, name, role) → Character

Adds a new character to an existing story.

python
hero = toolkit.add_character_to_story(story, "Kai", "protagonist")
hero.add_trait("brave")
check_story_coherence(story_data) → dict

Checks story coherence and generates a detailed report.

python
report = toolkit.check_story_coherence(story)
print(f"Score: {report['overall_score']:.0%}")
generate_full_story(genre, theme, num_characters=3) → dict

Generates a complete story with all elements automatically.

python
full_story = toolkit.generate_full_story(
    "mystery", "justice", num_characters=4
)
export_story(story, format="json", filename=None) → str

Exports story to a file in the specified format.

python
path = toolkit.export_story(story, format="markdown", filename="my_story")
print(f"Saved to: {path}")

python
from story_toolkit.core.story_engine import StoryEngine

Character

Character creation and management with full attribute system.

python
from story_toolkit.core.character import Character

# Create character
hero = Character(name="Elena", age=32, role="protagonist")

# Add traits
hero.add_trait("brave")
hero.add_trait("intelligent")

# Add goals
hero.add_goal("Save the kingdom")
hero.add_goal("Find the lost artifact")

# Add skills
hero.add_skill("sword_mastery")
hero.add_skill("stealth")
hero.add_skill("diplomacy")

# Add weaknesses
hero.add_weakness("overconfidence")
hero.add_weakness("impatience")

# Add fears
hero.add_fear("darkness")
hero.add_fear("failure")

# Add motivations
hero.add_motivation("justice")
hero.add_motivation("honor")

# Add relationships
hero.add_relationship("Villain", "enemy", strength=9)
hero.add_relationship("Mentor", "mentor", strength=10)
hero.add_relationship("Sidekick", "friend", strength=8)

# Create conflict
conflict = hero.add_conflict("internal")

# Character arc
print(hero.arc_stage)  # "initial"
hero.advance_arc()
print(hero.arc_stage)  # "challenged"

# Convert to dict
data = hero.to_dict()
MethodParametersDescription
add_trait(trait)strAdd personality trait
remove_trait(trait)strRemove personality trait
add_goal(goal)strAdd goal
add_skill(skill)strAdd skill
add_weakness(weakness)strAdd weakness
add_fear(fear)strAdd fear
add_motivation(motivation)strAdd motivation
add_relationship(name, type, strength=5)str, str, intAdd relationship: friend, enemy, family, mentor, ally, rival
update_relationship_status(name, status)str, strUpdate relationship status
add_conflict(type)strCreate conflict: internal, external, moral, relationship, societal
get_character_arc()—Get character arc
advance_arc()—Advance: initial → challenged → transformation → new_equilibrium
get_strengths_weaknesses()—Get strengths/weaknesses summary
to_dict()—Convert to dictionary

Plot

Plot structure management system.

python
from story_toolkit.core.plot import Plot

plot = Plot()

# Add plot points
point1 = plot.add_plot_point(
    "The Discovery",
    "Hero finds the ancient map",
    chapter=3,
    importance=8
)
point2 = plot.add_plot_point(
    "The Climax",
    "Final battle with the villain",
    chapter=18,
    importance=10
)

# Connect plot points
plot.connect_plot_points(point1.id, point2.id)

# Add subplot
subplot = plot.add_subplot(
    "Forbidden Love",
    "Hero falls for enemy's daughter",
    connected_characters=["Hero", "Princess"]
)

# Get timeline
timeline = plot.get_plot_timeline()
for event in timeline:
    print(f"Ch.{event['chapter']}: {event['title']}")

# Validate
issues = plot.validate_plot()
MethodParametersDescription
add_plot_point(title, desc, chapter, importance)str, str, int, intAdd plot point
add_subplot(title, desc, characters)str, str, listAdd subplot
connect_plot_points(id1, id2)str, strConnect two plot points
get_plot_summary()—Get plot statistics
get_plot_timeline()—Get chronological timeline
validate_plot()—Validate structure

WorldBuilder

World building and setting management system.

python
from story_toolkit.core.world_builder import WorldBuilder

world = WorldBuilder()

# Create world
world.create_world("Eldoria", "fantasy")

# Add locations
city = world.add_location("Crystal City", "Ancient metropolis", "city")
forest = world.add_location("Shadow Forest", "Mysterious forest", "forest")

# Connect locations
world.connect_locations("Crystal City", "Shadow Forest")

# Add rules
world.add_rule("physical", "Magic flows through crystal veins")
world.add_rule("magical", "Only eclipse-born can wield magic")
world.add_rule("social", "Council of Seven governs all")

# Add culture
culture = world.add_culture(
    "Crystal Weavers",
    "Ancient crystal-wielding society",
    customs=["crystal_carving", "meditation"]
)

# Add faction
faction = world.add_faction(
    "Shadow Guild",
    "Secret underground organization",
    goals=["control_magic", "overthrow_council"]
)

# Auto-generate world template
template = world.generate_world("fantasy")

# Get summary
summary = world.get_world_summary()
MethodParametersDescription
create_world(name, type)str, strInitialize new world
add_location(name, desc, type)str, str, strAdd location: city, forest, mountains, underground
connect_locations(loc1, loc2)str, strConnect two locations
add_rule(category, rule)str, strAdd rule: physical, magical, social, technological
add_culture(name, desc, customs)str, str, listAdd culture
add_faction(name, desc, goals)str, str, listAdd faction
generate_world(genre)strAuto-generate world template
get_world_summary()—Get world statistics

CharacterGenerator

Generates random or templated characters.

python
from story_toolkit.generators.character_generator import CharacterGenerator

gen = CharacterGenerator()

# Generate single character
hero = gen.generate_character("protagonist", complexity=4)
print(hero.name)             # Random name like "Alexander Storm"
print(hero.personality_traits)  # ["brave", "determined", ...]

# Generate ensemble
ensemble = gen.generate_ensemble(4)
for char in ensemble:
    print(f"{char.name} - {char.role}")

# Generate with specific backstory
hero2 = gen.generate_with_backstory("protagonist", "redemption")
print(hero2.background)
MethodParametersDescription
generate_character(role, complexity)str, intGenerate random character
generate_ensemble(num)intGenerate balanced ensemble
generate_with_backstory(role, theme)str, strGenerate with backstory: redemption, discovery, legacy, survival

PlotGenerator

Generates plot structures for various genres.

python
from story_toolkit.generators.plot_generator import PlotGenerator

gen = PlotGenerator()
plot = gen.generate_plot("mystery", complexity=4)

for stage in plot['main_plot']:
    print(f"{stage['stage']}: {stage['description']}")

print(f"Chapters: {plot['estimated_length']['estimated_chapters']}")
print(f"Words: {plot['estimated_length']['estimated_words']:,}")

DialogueGenerator

Generates natural-sounding dialogues between characters.

python
from story_toolkit.generators.dialogue_generator import DialogueGenerator

gen = DialogueGenerator()

# Generate dialogue
dialogue = gen.generate_dialogue(
    "Hero", "Villain",
    context="conflict",
    mood="tense"
)

# Create full scene
scene = gen.create_conversation_scene(
    [{"name": "Elena", "role": "protagonist"},
     {"name": "Victor", "role": "antagonist"}],
    "confrontation"
)
print(scene["setting"])     # "an abandoned warehouse in the rain"
print(scene["atmosphere"])  # "charged and volatile"

# Generate monologue
monologue = gen.generate_monologue("Hero", "destiny", "reflective")
MethodParametersDescription
generate_dialogue(char1, char2, context, mood)str, str, str, strGenerate dialogue. context: conflict, revelation, emotional, romantic, mysterious
create_conversation_scene(chars, purpose)list, strCreate full scene. purpose: confession, confrontation, romantic, mystery
generate_monologue(char, topic, mood)str, str, strGenerate monologue. mood: reflective, determined

CoherenceChecker

Analyzes story coherence and identifies narrative issues.

python
from story_toolkit.nlp.coherence_checker import CoherenceChecker

checker = CoherenceChecker()

# Full report
report = checker.generate_coherence_report(story_data)

# Quick check (True/False)
is_ok = checker.quick_check(story_data)

# Detailed issues
details = checker.get_detailed_issues(story_data)

TextAnalyzer

Analyzes text characteristics for readability, pacing, and dialogue.

python
from story_toolkit.nlp.text_analyzer import TextAnalyzer

analyzer = TextAnalyzer()

# Analyze text
text = "The old house stood on the hill."
analysis = analyzer.analyze_text(text)
print(analysis["readability_score"])

# Readability only
readability = analyzer.analyze_readability(text)

# Dialogue analysis
dialogue_analysis = analyzer.analyze_dialogue([
    "Hero: I will stop you!",
    "Villain: You can try!"
])

# Pacing analysis
pacing = analyzer.analyze_pacing(long_text, chunk_size=100)
