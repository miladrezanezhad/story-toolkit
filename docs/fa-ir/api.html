مستندات API

مرجع کامل تمام کلاس‌ها، متدها و پارامترهای Story Development Toolkit نسخه ۲.۰.۰

جدید در نسخه ۲.۰.۰: لایه اختیاری LLM با پشتیبانی از OpenAI، Anthropic و مدل‌های محلی!
Python 3.11+ نسخه ۲.۰.۰ MIT آماده LLM

نصب

bash 
git clone https://github.com/miladrezanezhad/story-toolkit.git
cd story-toolkit
pip install -r requirements.txt

# جدید در نسخه ۲.۰.۰ - نصب با پشتیبانی از LLM
pip install story-toolkit[openai]      # پشتیبانی از OpenAI
pip install story-toolkit[anthropic]   # پشتیبانی از Anthropic
pip install story-toolkit[local]       # پشتیبانی از مدل‌های محلی
pip install story-toolkit[all]         # همه پشتیبان‌ها

لایه LLM جدید در نسخه ۲.۰.۰

یکپارچگی اختیاری با مدل‌های زبانی بزرگ برای تولید دیالوگ پیشرفته و بهبود متن.

LLMProvider (ارائه‌دهندگان)

ارائه‌دهندهمقدارتوضیح
OPENAI"openai"مدل‌های OpenAI (GPT-4، GPT-3.5)
ANTHROPIC"anthropic"مدل‌های Anthropic (Claude)
LOCAL"local"مدل‌های محلی (Ollama، llama.cpp)
MOCK"mock"پشتیبان تست (بدون نیاز به کلید API)

LLMConfig

python
from story_toolkit.llm import LLMConfig, LLMProvider

config = LLMConfig(
    provider=LLMProvider.OPENAI,
    model="gpt-3.5-turbo",
    temperature=0.8,
    max_tokens=1000,
    api_key="sk-..."
)

LLMFactory

python
from story_toolkit.llm import LLMFactory, LLMProvider

# ایجاد پشتیبان Mock (بدون نیاز به کلید API - برای تست)
llm = LLMFactory.create_backend(provider=LLMProvider.MOCK)

# ایجاد پشتیبان OpenAI
llm = LLMFactory.create_backend(
    provider=LLMProvider.OPENAI,
    api_key="sk-...",
    model="gpt-3.5-turbo",
    temperature=0.8
)

# ایجاد پشتیبان محلی (Ollama)
llm = LLMFactory.create_backend(
    provider=LLMProvider.LOCAL,
    model="llama2",
    temperature=0.7
)

# ایجاد از متغیرهای محیطی
llm = LLMFactory.create_from_env()

StoryToolkit

کلاس اصلی که دسترسی یکپارچه به تمام امکانات toolkit را فراهم می‌کند.

سازنده (Constructor)

python 
from story_toolkit import StoryToolkit

# حالت معمولی (سازگار با نسخه ۱)
toolkit = StoryToolkit()

# جدید در نسخه ۲.۰.۰ - با پشتیبان LLM
toolkit = StoryToolkit(llm_backend=llm)

ویژگی‌ها (Attributes)

ویژگینوعتوضیح
engineStoryEngineموتور اصلی داستان
character_genCharacterGeneratorتولید شخصیت تصادفی
plot_genPlotGeneratorتولید ساختار پیرنگ
dialogue_genDialogueGeneratorتولید دیالوگ
coherence_checkerCoherenceCheckerبررسی انسجام داستان
text_analyzerTextAnalyzerتحلیل متن و خوانایی

متدها

create_story(genre, theme, complexity=3) → dict

یک داستان جدید با تمام اجزای لازم ایجاد می‌کند.

:
پارامترنوعپیش‌فرضتوضیح
genrestrالزامیژانر: fantasy, mystery, romance, adventure, sci_fi
themestrالزامیتم اصلی مانند courage, redemption, identity
complexityint3سطح پیچیدگی از ۱ تا ۵

خروجی: دیکشنری با کلیدهای metadata، outline، plot، characters، world، dialogue_scenes، coherence_report

python
story = toolkit.create_story(genre="fantasy", theme="courage", complexity=4)
print(story["metadata"]["genre"])  # "fantasy"
add_character_to_story(story, name, role) → Character

یک شخصیت جدید به داستان اضافه می‌کند.

python
hero = toolkit.add_character_to_story(story, "کای", "protagonist")
hero.add_trait("brave")
check_story_coherence(story_data) → dict

انسجام داستان را بررسی و گزارش کامل تولید می‌کند.

python
report = toolkit.check_story_coherence(story)
print(f"امتیاز: {report['overall_score']:.0%}")
generate_full_story(genre, theme, num_characters=3, use_advanced_dialogue=False) → dict

یک داستان کامل با تمام عناصر به صورت خودکار تولید می‌کند.

:
پارامترنوعپیش‌فرضتوضیح
genrestrالزامیژانر داستان
themestrالزامیتم داستان
num_charactersint3تعداد شخصیت‌ها
use_advanced_dialogueboolFalseجدید استفاده از LLM برای دیالوگ
python
# حالت معمولی
full_story = toolkit.generate_full_story("mystery", "justice", num_characters=4)

# جدید در نسخه ۲.۰.۰ - با دیالوگ پیشرفته
full_story = toolkit.generate_full_story(
    "mystery", "justice",
    num_characters=4,
    use_advanced_dialogue=True
)
export_story(story, format="json", filename=None) → str

داستان را در قالب مشخص خروجی می‌گیرد.

path = toolkit.export_story(story, format="markdown", filename="my_story")

Character

سیستم ایجاد و مدیریت شخصیت با ویژگی‌های کامل.

python
from story_toolkit.core.character import Character

# ایجاد شخصیت
hero = Character(name="النا", age=32, role="protagonist")

# افزودن ویژگی‌ها
hero.add_trait("brave")         # شجاع
hero.add_trait("intelligent")   # باهوش

# افزودن اهداف
hero.add_goal("Save the kingdom")
hero.add_goal("Find the lost artifact")

# افزودن مهارت‌ها
hero.add_skill("sword_mastery")
hero.add_skill("stealth")
hero.add_skill("diplomacy")

# افزودن نقاط ضعف
hero.add_weakness("overconfidence")
hero.add_weakness("impatience")

# افزودن ترس‌ها
hero.add_fear("darkness")
hero.add_fear("failure")

# افزودن انگیزه‌ها
hero.add_motivation("justice")
hero.add_motivation("honor")

# افزودن روابط
hero.add_relationship("Villain", "enemy", strength=9)
hero.add_relationship("Mentor", "mentor", strength=10)
hero.add_relationship("Sidekick", "friend", strength=8)

# ایجاد تعارض
conflict = hero.add_conflict("internal")

# قوس شخصیتی
print(hero.arc_stage)  # "initial"
hero.advance_arc()
print(hero.arc_stage)  # "challenged"

# تبدیل به دیکشنری
data = hero.to_dict()
:
متدپارامترهاتوضیح
add_trait(trait)strافزودن ویژگی شخصیتی
remove_trait(trait)strحذف ویژگی شخصیتی
add_goal(goal)strافزودن هدف
add_skill(skill)strافزودن مهارت
add_weakness(weakness)strافزودن نقطه ضعف
add_fear(fear)strافزودن ترس
add_motivation(motivation)strافزودن انگیزه
add_relationship(name, type, strength=5)str, str, intافزودن رابطه: friend, enemy, family, mentor, ally, rival
update_relationship_status(name, status)str, strبه‌روزرسانی وضعیت رابطه
add_conflict(type)strایجاد تعارض: internal, external, moral, relationship, societal
get_character_arc()—دریافت قوس شخصیتی
advance_arc()—پیشبرد: initial → challenged → transformation → new_equilibrium
get_strengths_weaknesses()—خلاصه نقاط قوت و ضعف
to_dict()—تبدیل به دیکشنری

Plot

سیستم مدیریت ساختار پیرنگ داستان.

python
from story_toolkit.core.plot import Plot

plot = Plot()

# افزودن نقاط عطف
point1 = plot.add_plot_point(
    "The Discovery",
    "Hero finds the ancient map",
    chapter=3,
    importance=8
)
point2 = plot.add_plot_point(
    "The Climax",
    "Final battle with the villain",
    chapter=18,
    importance=10
)

# اتصال نقاط
plot.connect_plot_points(point1.id, point2.id)

# افزودن زیرپیرنگ
subplot = plot.add_subplot(
    "Forbidden Love",
    "Hero falls for enemy's daughter",
    connected_characters=["Hero", "Princess"]
)

# دریافت جدول زمانی
timeline = plot.get_plot_timeline()
for event in timeline:
    print(f"Ch.{event['chapter']}: {event['title']}")

# اعتبارسنجی
issues = plot.validate_plot()
:
متدپارامترهاتوضیح
add_plot_point(title, desc, chapter, importance)str, str, int, intافزودن نقطه عطف
add_subplot(title, desc, characters)str, str, listافزودن زیرپیرنگ
connect_plot_points(id1, id2)str, strاتصال دو نقطه عطف
get_plot_summary()—خلاصه آمار پیرنگ
get_plot_timeline()—جدول زمانی وقایع
validate_plot()—اعتبارسنجی ساختار

DialogueGenerator

تولید دیالوگ‌های طبیعی بین شخصیت‌ها با پشتیبانی اختیاری از LLM.

سازنده (Constructor)

python
# حالت معمولی (سازگار با نسخه ۱)
gen = DialogueGenerator()

# جدید در نسخه ۲.۰.۰ - با پشتیبانی LLM
gen = DialogueGenerator(llm_backend=llm)

متد generate_dialogue (بروز شده)

python
# حالت معمولی (بدون LLM)
dialogue = gen.generate_dialogue("قهرمان", "شرور", context="conflict")

# جدید - حالت پیشرفته (با LLM)
dialogue = gen.generate_dialogue(
    "قهرمان", "شرور",
    context="نبرد نهایی",
    use_advanced=True,      # فعال‌سازی LLM
    style="دراماتیک",        # natural, dramatic, poetic, humorous
    num_lines=10
)
