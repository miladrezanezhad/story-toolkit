StoryToolkit

کلاس اصلی که دسترسی یکپارچه به تمام امکانات toolkit را فراهم می‌کند.

سازنده (Constructor)

python 
from story_toolkit import StoryToolkit

toolkit = StoryToolkit()

ویژگی‌ها (Attributes)

ویژگینوعتوضیح
engineStoryEngineموتور اصلی داستان
character_genCharacterGeneratorتولید شخصیت تصادفی
plot_genPlotGeneratorتولید ساختار پیرنگ
dialogue_genDialogueGeneratorتولید دیالوگ
coherence_checkerCoherenceCheckerبررسی انسجام داستان
text_analyzerTextAnalyzerتحلیل متن و خوانایی

متدها

create_story(genre, theme, complexity=3) → dict

یک داستان جدید با تمام اجزای لازم ایجاد می‌کند.

پارامترنوعپیش‌فرضتوضیح
genrestrالزامیژانر: fantasy, mystery, romance, adventure, sci_fi
themestrالزامیتم اصلی مانند courage, redemption, identity
complexityint3سطح پیچیدگی از ۱ تا ۵

خروجی: دیکشنری با کلیدهای metadata، outline، plot، characters، world، dialogue_scenes، coherence_report

python
story = toolkit.create_story(genre="fantasy", theme="courage", complexity=4)
print(story["metadata"]["genre"])  # "fantasy"
add_character_to_story(story, name, role) → Character

یک شخصیت جدید به داستان اضافه می‌کند.

python
hero = toolkit.add_character_to_story(story, "کای", "protagonist")
hero.add_trait("brave")
check_story_coherence(story_data) → dict

انسجام داستان را بررسی و گزارش کامل تولید می‌کند.

python
report = toolkit.check_story_coherence(story)
print(f"امتیاز: {report['overall_score']:.0%}")
generate_full_story(genre, theme, num_characters=3) → dict

یک داستان کامل با تمام عناصر به صورت خودکار تولید می‌کند.

python
full_story = toolkit.generate_full_story(
    "mystery", "justice", num_characters=4
)
export_story(story, format="json", filename=None) → str

داستان را در قالب مشخص خروجی می‌گیرد.

python
path = toolkit.export_story(story, format="markdown", filename="my_story")
print(f"ذخیره شد: {path}")

Character

سیستم ایجاد و مدیریت شخصیت با ویژگی‌های کامل.

python
from story_toolkit.core.character import Character

# ایجاد شخصیت
hero = Character(name="النا", age=32, role="protagonist")

# افزودن ویژگی‌ها
hero.add_trait("brave")         # شجاع
hero.add_trait("intelligent")   # باهوش

# افزودن اهداف
hero.add_goal("Save the kingdom")
hero.add_goal("Find the lost artifact")

# افزودن مهارت‌ها
hero.add_skill("sword_mastery")
hero.add_skill("stealth")
hero.add_skill("diplomacy")

# افزودن نقاط ضعف
hero.add_weakness("overconfidence")
hero.add_weakness("impatience")

# افزودن ترس‌ها
hero.add_fear("darkness")
hero.add_fear("failure")

# افزودن انگیزه‌ها
hero.add_motivation("justice")
hero.add_motivation("honor")

# افزودن روابط
hero.add_relationship("Villain", "enemy", strength=9)
hero.add_relationship("Mentor", "mentor", strength=10)
hero.add_relationship("Sidekick", "friend", strength=8)

# ایجاد تعارض
conflict = hero.add_conflict("internal")
print(conflict)  # "النا struggles with inner doubts and fears"

# قوس شخصیتی
print(hero.arc_stage)  # "initial"
hero.advance_arc()
print(hero.arc_stage)  # "challenged"

# تبدیل به دیکشنری
data = hero.to_dict()
متدپارامترهاتوضیح
add_trait(trait)strافزودن ویژگی شخصیتی
remove_trait(trait)strحذف ویژگی شخصیتی
add_goal(goal)strافزودن هدف
add_skill(skill)strافزودن مهارت
add_weakness(weakness)strافزودن نقطه ضعف
add_fear(fear)strافزودن ترس
add_motivation(motivation)strافزودن انگیزه
add_relationship(name, type, strength=5)str, str, intافزودن رابطه: friend, enemy, family, mentor, ally, rival
update_relationship_status(name, status)str, strبه‌روزرسانی وضعیت رابطه
add_conflict(type)strایجاد تعارض: internal, external, moral, relationship, societal
get_character_arc()—دریافت قوس شخصیتی
advance_arc()—پیشبرد: initial → challenged → transformation → new_equilibrium
get_strengths_weaknesses()—خلاصه نقاط قوت و ضعف
to_dict()—تبدیل به دیکشنری

Plot

سیستم مدیریت ساختار پیرنگ داستان.

python
from story_toolkit.core.plot import Plot

plot = Plot()

# افزودن نقاط عطف
point1 = plot.add_plot_point(
    "The Discovery",
    "Hero finds the ancient map",
    chapter=3,
    importance=8
)
point2 = plot.add_plot_point(
    "The Climax",
    "Final battle with the villain",
    chapter=18,
    importance=10
)

# اتصال نقاط
plot.connect_plot_points(point1.id, point2.id)

# افزودن زیرپیرنگ
subplot = plot.add_subplot(
    "Forbidden Love",
    "Hero falls for enemy's daughter",
    connected_characters=["Hero", "Princess"]
)

# دریافت جدول زمانی
timeline = plot.get_plot_timeline()
for event in timeline:
    print(f"Ch.{event['chapter']}: {event['title']}")

# اعتبارسنجی
issues = plot.validate_plot()
if issues:
    for issue in issues:
        print(f"⚠️ {issue}")
متدپارامترهاتوضیح
add_plot_point(title, desc, chapter, importance)str, str, int, intافزودن نقطه عطف
add_subplot(title, desc, characters)str, str, listافزودن زیرپیرنگ
connect_plot_points(id1, id2)str, strاتصال دو نقطه عطف
get_plot_summary()—خلاصه آمار پیرنگ
get_plot_timeline()—جدول زمانی وقایع
validate_plot()—اعتبارسنجی ساختار

WorldBuilder

سیستم ساخت و مدیریت جهان داستان.

python
from story_toolkit.core.world_builder import WorldBuilder

world = WorldBuilder()

# ایجاد جهان
world.create_world("الدوریا", "fantasy")

# افزودن مکان‌ها
city = world.add_location("شهر کریستال", "کلانشهر باستانی", "city")
forest = world.add_location("جنگل سایه‌ها", "جنگل مرموز", "forest")

# اتصال مکان‌ها
world.connect_locations("شهر کریستال", "جنگل سایه‌ها")

# افزودن قوانین
world.add_rule("physical", "جادو از طریق کریستال‌ها جریان دارد")
world.add_rule("magical", "فقط متولدان ماه‌گرفتگی جادو دارند")
world.add_rule("social", "شورای هفت نفر بر امور جادویی حکومت می‌کند")

# افزودن فرهنگ
culture = world.add_culture(
    "کریستال‌بافان",
    "جامعه باستانی استفاده از کریستال",
    customs=["کریستال‌تراشی", "مراقبه"]
)

# افزودن گروه
faction = world.add_faction(
    "انجمن سایه",
    "سازمان مخفی زیرزمینی",
    goals=["کنترل جادو", "سرنگونی شورا"]
)

# تولید خودکار جهان براساس ژانر
template = world.generate_world("fantasy")

# خلاصه جهان
summary = world.get_world_summary()
print(summary["total_locations"])
متدپارامترهاتوضیح
create_world(name, type)str, strایجاد جهان جدید
add_location(name, desc, type)str, str, strافزودن مکان: city, forest, mountains, underground
connect_locations(loc1, loc2)str, strاتصال دو مکان
add_rule(category, rule)str, strافزودن قانون: physical, magical, social, technological
add_culture(name, desc, customs)str, str, listافزودن فرهنگ
add_faction(name, desc, goals)str, str, listافزودن گروه
generate_world(genre)strتولید خودکار جهان
get_world_summary()—خلاصه آمار جهان

CharacterGenerator

تولید شخصیت‌های تصادفی یا قالبی.

python
from story_toolkit.generators.character_generator import CharacterGenerator

gen = CharacterGenerator()

# تولید یک شخصیت
hero = gen.generate_character("protagonist", complexity=4)
print(hero.name)             # اسم تصادفی مثل "Alexander Storm"
print(hero.personality_traits)  # ["brave", "determined", ...]

# تولید گروه کامل
ensemble = gen.generate_ensemble(4)
for char in ensemble:
    print(f"{char.name} - {char.role}")

# تولید با پیش‌زمینه خاص
hero2 = gen.generate_with_backstory("protagonist", "redemption")
print(hero2.background)
متدپارامترهاتوضیح
generate_character(role, complexity)str, intتولید شخصیت تصادفی
generate_ensemble(num)intتولید گروه متعادل
generate_with_backstory(role, theme)str, strتولید با پیش‌زمینه: redemption, discovery, legacy, survival

PlotGenerator

تولید ساختار پیرنگ برای ژانرهای مختلف.

python
from story_toolkit.generators.plot_generator import PlotGenerator

gen = PlotGenerator()

# تولید پیرنگ
plot = gen.generate_plot("mystery", complexity=4)

# نمایش مراحل
for stage in plot['main_plot']:
    print(f"{stage['stage']}: {stage['description']}")

# پیچش‌های داستانی
for twist in plot['twists']:
    print(f"🌀 {twist}")

# برآورد طول داستان
print(f"فصل‌ها: {plot['estimated_length']['estimated_chapters']}")
print(f"کلمات: {plot['estimated_length']['estimated_words']:,}")

DialogueGenerator

تولید دیالوگ‌های طبیعی بین شخصیت‌ها.

python
from story_toolkit.generators.dialogue_generator import DialogueGenerator

gen = DialogueGenerator()

# دیالوگ ساده
dialogue = gen.generate_dialogue(
    "Hero", "Villain",
    context="conflict",
    mood="tense"
)
for line in dialogue:
    print(line)

# صحنه کامل
scene = gen.create_conversation_scene(
    [
        {"name": "Elena", "role": "protagonist"},
        {"name": "Victor", "role": "antagonist"}
    ],
    "confrontation"
)
print(scene["setting"])      # "an abandoned warehouse in the rain"
print(scene["atmosphere"])   # "charged and volatile"

# مونولوگ
monologue = gen.generate_monologue("Hero", "destiny", "reflective")
متدپارامترهاتوضیح
generate_dialogue(char1, char2, context, mood)str, str, str, strتولید دیالوگ. context: conflict, revelation, emotional, romantic, mysterious
create_conversation_scene(chars, purpose)list, strایجاد صحنه کامل. purpose: confession, confrontation, romantic, mystery
generate_monologue(char, topic, mood)str, str, strتولید مونولوگ. mood: reflective, determined

TextAnalyzer

تحلیل ویژگی‌های متن، خوانایی و ریتم.

python
from story_toolkit.nlp.text_analyzer import TextAnalyzer

analyzer = TextAnalyzer()

# تحلیل متن
text = "The old house stood on the hill. It had been abandoned for years."
analysis = analyzer.analyze_text(text)
print(analysis["readability_score"])

# تحلیل خوانایی
readability = analyzer.analyze_readability(text)

# تحلیل دیالوگ
dialogue_analysis = analyzer.analyze_dialogue([
    "Hero: I will stop you!",
    "Villain: You can try!"
])

# تحلیل ریتم
pacing = analyzer.analyze_pacing(long_text, chunk_size=100)
