شروع سریع

در ۵ دقیقه اولین داستان خود را با Story Development Toolkit بسازید

مرحله ۱: نصب

bash 
git clone https://github.com/miladrezanezhad/story-toolkit.git
cd story-toolkit
pip install -r requirements.txt

یا نصب مستقیم با pip install -e .

مرحله ۲: import کتابخانه

python 
from story_toolkit import StoryToolkit

# ایجاد نمونه از Toolkit
toolkit = StoryToolkit()

کلاس StoryToolkit دسترسی یکپارچه به تمام امکانات را فراهم می‌کند.

مرحله ۳: ساخت داستان

python 
# ایجاد یک داستان فانتزی با تم شجاعت
story = toolkit.create_story(
    genre="fantasy",
    theme="courage",
    complexity=4
)

print(story["metadata"]["genre"])   # "fantasy"
print(story["metadata"]["theme"])   # "courage"
print(story["metadata"]["complexity"])  # 4
نکته: ژانرهای پشتیبانی شده: fantasy، mystery، romance، adventure، sci_fi

مرحله ۴: افزودن شخصیت‌ها

python 
# افزودن قهرمان داستان
hero = toolkit.add_character_to_story(
    story,
    "کای",
    "protagonist"
)

# توسعه شخصیت قهرمان
hero.add_trait("brave")         # شجاع
hero.add_trait("determined")     # مصمم
hero.add_trait("compassionate")  # دلسوز
hero.add_goal("Save the kingdom")   # نجات پادشاهی
hero.add_skill("sword_mastery")    # شمشیرزنی
hero.add_skill("stealth")         # حرکت بی‌صدا

# افزودن ضدقهرمان
villain = toolkit.add_character_to_story(
    story,
    "مورگان",
    "antagonist"
)
villain.add_trait("manipulative")  # حیله‌گر
villain.add_trait("charismatic")   # کاریزماتیک
villain.add_goal("Control the kingdom")  # تسخیر پادشاهی

# ایجاد رابطه بین شخصیت‌ها
hero.add_relationship("مورگان", "enemy", strength=9)
نقش‌های قابل استفاده: protagonist (قهرمان)، antagonist (ضدقهرمان)، supporting (مکمل)، mentor (راهنما)

مرحله ۵: تولید دیالوگ

python 
# دیالوگ درگیری بین قهرمان و ضدقهرمان
dialogue = toolkit.dialogue_gen.generate_dialogue(
    "کای",
    "مورگان",
    context="conflict",
    mood="tense"
)

for line in dialogue:
    print(line)
خروجی:

کای: نمی‌تونم باور کنم این کار رو کردی!
مورگان: چاره‌ای برام نذاشتی.
کای: همیشه یه راه هست. تو فقط اشتباه انتخاب کردی.
زمینه‌های دیالوگ: conflict (درگیری)، revelation (افشاگری)، emotional (عاطفی)، romantic (عاشقانه)، mysterious (مرموز)

مرحله ۶: بررسی انسجام داستان

python 
# بررسی انسجام و پیوستگی داستان
report = toolkit.check_story_coherence(story)

# نمایش امتیاز کلی
print(f"امتیاز انسجام: {report['overall_score']:.0%}")

# بررسی حفره‌های داستانی
if report['plot_holes']:
    print("\n⚠️ حفره‌های داستانی:")
    for hole in report['plot_holes']:
        print(f"  - {hole}")

# نمایش پیشنهادات بهبود
if report['recommendations']:
    print("\n💡 پیشنهادات:")
    for rec in report['recommendations']:
        print(f"  - {rec}")
خروجی:

امتیاز انسجام: 100%

💡 پیشنهادات:
- Story appears coherent - maintain this quality throughout!

مرحله ۷: ذخیره و خروجی

python 
from story_toolkit.utils.helpers import save_story, export_to_markdown

# ذخیره به فرمت JSON
save_story(story, "my_story.json")

# خروجی Markdown برای نمایش در گیت‌هاب
export_to_markdown(story, "my_story.md")

# یا استفاده از متد export خود Toolkit
toolkit.export_story(story, format="json", filename="my_story")
toolkit.export_story(story, format="markdown", filename="my_story")

print("✅ داستان با موفقیت ذخیره شد!")

🎯 کد کامل یک‌جا

python 
from story_toolkit import StoryToolkit
from story_toolkit.utils.helpers import save_story

# ۱. ایجاد نمونه
toolkit = StoryToolkit()

# ۲. ساخت داستان
story = toolkit.create_story("fantasy", "courage", complexity=4)

# ۳. افزودن شخصیت‌ها
hero = toolkit.add_character_to_story(story, "کای", "protagonist")
hero.add_trait("brave")
hero.add_goal("Save the kingdom")

villain = toolkit.add_character_to_story(story, "مورگان", "antagonist")
villain.add_trait("manipulative")

# ۴. ایجاد رابطه
hero.add_relationship("مورگان", "enemy", strength=9)

# ۵. تولید دیالوگ
dialogue = toolkit.dialogue_gen.generate_dialogue(
    "کای", "مورگان", context="conflict"
)
for line in dialogue:
    print(line)

# ۶. بررسی انسجام
report = toolkit.check_story_coherence(story)
print(f"\n✅ امتیاز انسجام: {report['overall_score']:.0%}")

# ۷. ذخیره داستان
save_story(story, "my_story.json")
print("📁 داستان در my_story.json ذخیره شد")
